To create a DNS zone:
  kubectl apply -f - <<EOF
  apiVersion: dns.contentways.org/v1alpha1
  kind: DNSZone
  metadata:
    name: my-zone
    namespace: {{ .Release.Namespace }}
  spec:
    name: example.org
    type: NATIVE
    nameservers:
      - ns1.example.org
      - ns2.example.org
  EOF

To create a DNS record:
  kubectl apply -f - <<EOF
  apiVersion: dns.contentways.org/v1alpha1
  kind: DNSRecord
  metadata:
    name: www-example
    namespace: {{ .Release.Namespace }}
  spec:
    zoneName: example.org
    name: www
    type: A
    content: 1.2.3.4
    ttl: 3600
  EOF

Check zone status:
  kubectl get dnszones
  kubectl get dnsrecords
